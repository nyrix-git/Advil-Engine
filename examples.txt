// Game: Apple Clicker
import org.newdawn.slick.Color;
import org.newdawn.slick.GameContainer;
import org.newdawn.slick.Graphics;
import org.newdawn.slick.Input;
import org.newdawn.slick.geom.Rectangle;

import com.AdvilEngine.core.AdvilEngine;
import com.AdvilEngine.helperutil.HelperUtils;
import com.AdvilEngine.input.InputManager;
import com.AdvilEngine.window.WindowManager;

public class MainExec extends AdvilEngine {

	public static int Score = 0;

	// cursor properties

	private static int Cursor_xPosition;
	private static int Cursor_yPosition;

	// apple properties
	private static int Apple_xPosition;
	private static int Apple_yPosition;

	private static Rectangle appleRectangle = new Rectangle(Apple_xPosition, Apple_yPosition, 200, 200);

	// random properties
	private static int DistanceFromEdgeScreen = 200;

	// window properties
	public static int windowWidth = 1280;
	public static int windowHeight = 720;
	private static String windowTitle = "Apple Clicker";

	public MainExec(String windowTitle_String) {
		super(windowTitle_String);
	}

	@Override
	public void Init(GameContainer container) {
		Apple_xPosition = HelperUtils.randomInt(0, MainExec.windowWidth - DistanceFromEdgeScreen);
		Apple_yPosition = HelperUtils.randomInt(0, MainExec.windowHeight - DistanceFromEdgeScreen);

		appleRectangle.setX(Apple_xPosition);
		appleRectangle.setY(Apple_yPosition);
	}

	@Override
	public void Tick(GameContainer container) {
		Cursor_xPosition = InputManager.getCursorX(container);
		Cursor_yPosition = InputManager.getCursorY(container);

		if (InputManager.mouseInputPressed(container, Input.MOUSE_LEFT_BUTTON)) {
			if (appleRectangle.contains(Cursor_xPosition, Cursor_yPosition)) {
				Score++;

				Apple_xPosition = HelperUtils.randomInt(0, MainExec.windowWidth - DistanceFromEdgeScreen);
				Apple_yPosition = HelperUtils.randomInt(0, MainExec.windowHeight - DistanceFromEdgeScreen);

				appleRectangle.setX(Apple_xPosition);
				appleRectangle.setY(Apple_yPosition);
			}
		}
	}

	@Override
	public void Render(Graphics g) {
		g.setColor(Color.red);
		g.fill(appleRectangle);
		g.setColor(Color.white);
		g.drawString("Score: " + Score, windowWidth / 2, 0);
	}

	public static void main(String[] args) throws Exception {
		MainExec mainexec = new MainExec(windowTitle);
		WindowManager.createWindow(mainexec, windowWidth, windowHeight, false);
	}
}

// Player movement
import org.newdawn.slick.GameContainer;
import org.newdawn.slick.Graphics;
import org.newdawn.slick.Input;
import org.newdawn.slick.geom.Rectangle;

import com.AdvilEngine.core.AdvilEngine;
import com.AdvilEngine.input.InputManager;
import com.AdvilEngine.window.WindowManager;

public class MainExec extends AdvilEngine {

	// window properties
	public static int windowWidth = 1280;
	public static int windowHeight = 720;
	private static String windowTitle = "Player Movement";
	private static boolean windowIsResizable = false;

	// player properties
	private static float playerMovementSpeed = 5f;

	// initial starting positions for player
	private static int Player_xPosition = windowWidth / 2;
	private static int Player_yPosition = windowHeight / 2;

	private static Rectangle playerRectangle = new Rectangle(Player_xPosition, Player_yPosition, 200, 200);

	public MainExec(String windowTitle_String) {
		super(windowTitle_String);
	}

	@Override
	public void Init(GameContainer container) {

	}

	@Override
	public void Tick(GameContainer container) {
		// w
		if (InputManager.keyboardInputDown(container, Input.KEY_W)) {
			Player_yPosition -= playerMovementSpeed;
		}

		// s
		if (InputManager.keyboardInputDown(container, Input.KEY_S)) {
			Player_yPosition += playerMovementSpeed;
		}

		// a
		if (InputManager.keyboardInputDown(container, Input.KEY_A)) {
			Player_xPosition -= playerMovementSpeed;
		}

		// d
		if (InputManager.keyboardInputDown(container, Input.KEY_D)) {
			Player_xPosition += playerMovementSpeed;
		}

		playerRectangle.setX(Player_xPosition);
		playerRectangle.setY(Player_yPosition);
	}

	@Override
	public void Render(Graphics g) {
		g.fill(playerRectangle);
	}

	public static void main(String[] args) throws Exception {
		MainExec mainexec = new MainExec(windowTitle);
		WindowManager.createWindow(mainexec, windowWidth, windowHeight, windowIsResizable);
	}

}

// title screen appear
import org.newdawn.slick.GameContainer;
import org.newdawn.slick.Graphics;
import org.newdawn.slick.Input;
import org.newdawn.slick.geom.Circle;

import com.AdvilEngine.core.AdvilEngine;
import com.AdvilEngine.input.InputManager;
import com.AdvilEngine.window.WindowManager;

public class MainExec extends AdvilEngine {

	// window properties
	public static int windowWidth = 1280;
	public static int windowHeight = 720;
	private static String windowTitle = "Title Screen Appear";
	private static boolean windowIsResizable = false;

	// title screen appearing properties
	private int APPEAR_TIME = 60 * 3; // 180 ticks = 3 seconds
	private int APPEAR_Timer = 0;
	
	private boolean isLogoRendered = true;
	private float logoScale = 10f;
	
	private Circle logoCircle = new Circle(windowWidth / 2, windowHeight / 2, logoScale);

	public MainExec(String windowTitle_String) {
		super(windowTitle_String);
	}

	@Override
	public void Init(GameContainer container) {

	}

	@Override
	public void Tick(GameContainer container) {
		logoScale++;
		// update the circle radius
		logoCircle.setRadius(logoScale);
		
		APPEAR_Timer++;
		if (APPEAR_Timer >= APPEAR_TIME) {
			isLogoRendered = false;
		}

		if (InputManager.keyboardInputPressed(container, Input.KEY_SPACE)) {
			APPEAR_Timer = APPEAR_TIME;
		}
	}

	@Override
	public void Render(Graphics g) {
		if (isLogoRendered == true) {
			g.fill(logoCircle);
		}
	}

	public static void main(String[] args) throws Exception {
		MainExec mainexec = new MainExec(windowTitle);
		WindowManager.createWindow(mainexec, windowWidth, windowHeight, windowIsResizable);
	}

}
