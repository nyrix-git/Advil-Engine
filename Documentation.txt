Camera: 

@ Create the camera

private CameraManager camera;

@Override // Java syntax
public void Init(GameContainer container) {
    camera = new CameraManager(0, 0); // creates camera on the coordinates (0, 0)
}

@ Move camera
camera.move(0, -5) // moves the camera up on the y axis

@ Apply the camera to rendering

@Override // Java syntax
public void Render(Graphics g) {

    // Apply camera transform
    camera.apply(g);

    // Draw everything affected by camera
    g.drawString("World Object", 500, 500);
    g.fillRect(400, 400, 50, 50);

    // Reset so the User Interface isn't affected by camera
    camera.reset(g);

    // Draw User Interface (this is fixed to the screen due to the camera not affecting this)
    g.drawString("HUD / UI", 10, 30);
}

Physics:

	Rectangle player = new Rectangle(100, 100, 32, 32);
	Rectangle ground = new Rectangle(0, 300, 800, 50);

	float velX = 0;
	float velY = 0;
	float friction = 0.2f;

	public void Tick() {
		// Apply gravity
    		velY = PhysicsManager.applyGravity(velY);

    		// Apply friction
    		velX = PhysicsManager.applyFriction(velX, friction);

    		// Save old positions
    		float oldX = player.getX();
    		float oldY = player.getY();

    		// Move X and check collision
    		boolean hitX = PhysicsManager.moveWithCollisionX(player, velX, ground);

    		// Move Y and check collision
    		boolean hitY = PhysicsManager.moveWithCollisionY(player, velY, ground);
    	}

	// change gravity
	PhysicsManager.setGravity(0.8f);
